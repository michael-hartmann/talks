    Parameter:
    \begin{itemize}

    \begin{itemize}

    }

    \end{itemize}

    \begin{itemize}
    \item 4 Freiheitsgrade:
    \begin{itemize}

    \item 2 Geschwindigkeiten $\dot \theta_1$, $\dot \theta_2$
    \end{itemize}
    \item Energieerhaltung: Trajektorien auf Hyperfläche eingeschränkt

    \only<1>
    {
    \begin{center}
    \includegraphics[scale=0.45]{images/poincare_idea.png}
    \includegraphics[scale=0.45]{images/poincare_cycle3.png}

    \hfill
    \hrule
    \ \\
    {\footnotesize Bildquelle: An Exploration of Chaos, Argyris, Faus, Haase}
    }

    \only<2>
    {
    Poincaré-Bedingung:
    $$\theta_2 = 0, \quad \dot\theta_2 > 0$$

    \vfill

    Poincaré-Schnitt:
    \begin{enumerate}
    \item Wähle Parameter $l$, $m$, $g$
    \item Wähle Energie $E$
    \item Wähle Anfangsbedingungen $\theta_1(0)$, $\dot\theta_1(0)$, $\theta_2(0)$, $\dot\theta_2(0)$
    \item Berechne numerisch $\theta_1(t)$, $\dot\theta_1(t)$, $\theta_2(t)$, $\dot\theta_2(t)$ und trage Punkt in Plot ein, wenn $\theta_2(t)=0$ und $\dot\theta_2 > 0$
    \end{enumerate}
    }

\frame {
    \frametitle{Zusammenfassung}

    Poincaré-Plot:
    \begin{itemize}
        \item Punkte: periodische Lösungen
        \item Linien: quasiperiodische Lösungen
        \item wild verstreute Punkte: Chaos
    \end{itemize}

    \vfill

    Chaos:
    \begin{itemize}
        \item System zwar deterministisch, aber...
        \item ...kleine Ursachen, große Wirkungen
        \item langfristige Vorhersagen nicht möglich
        \item Chaos ist nicht Zufall!
    \end{itemize}
}
